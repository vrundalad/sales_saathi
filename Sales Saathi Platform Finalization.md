  <script>
    document.addEventListener('DOMContentLoaded', () => {
      const toggleWrapper = document.getElementById('pricing-toggle');
      const btnMonthly = document.getElementById('btn-monthly');
      const btnYearly = document.getElementById('btn-yearly');
      const slider = document.getElementById('toggle-slider');
      const priceDisplays = document.querySelectorAll('.price-display');
      const yearlySavings = document.querySelectorAll('.yearly-savings');
      
      let isYearly = false;

      function updatePricing() {
        // Animate text fade out
        priceDisplays.forEach(el => el.classList.add('fade-out'));
        
        setTimeout(() => {
          // Update Values
          priceDisplays.forEach(el => {
            el.textContent = isYearly ? el.dataset.yearly : el.dataset.monthly;
            el.classList.remove('fade-out');
          });
          
          yearlySavings.forEach(el => {
            if (isYearly) {
              el.classList.remove('opacity-0');
            } else {
              el.classList.add('opacity-0');
            }
          });
        }, 150);
      }

      function setYearly() {
        if (isYearly) return;
        isYearly = true;
        
        // Update slider position (move right)
        slider.style.transform = `translateX(${btnMonthly.offsetWidth}px)`;
        slider.style.width = `${btnYearly.offsetWidth}px`;
        
        // Update text colors
        btnMonthly.classList.remove('text-white');
        btnMonthly.classList.add('text-gray-400');
        btnYearly.classList.remove('text-gray-400');
        btnYearly.classList.add('text-white');
        
        updatePricing();
      }

      function setMonthly() {
        if (!isYearly) return;
        isYearly = false;
        
        // Update slider position (move left)
        slider.style.transform = 'translateX(0)';
        slider.style.width = '120px'; // Original width of monthly button
        
        // Update text colors
        btnYearly.classList.remove('text-white');
        btnYearly.classList.add('text-gray-400');
        btnMonthly.classList.remove('text-gray-400');
        btnMonthly.classList.add('text-white');
        
        updatePricing();
      }

      // Event Listeners
      btnMonthly.addEventListener('click', (e) => {
        e.stopPropagation();
        setMonthly();
      });
      
      btnYearly.addEventListener('click', (e) => {
        e.stopPropagation();
        setYearly();
      });
      
      toggleWrapper.addEventListener('click', () => {
        if (isYearly) {
          setMonthly();
        } else {
          setYearly();
        }
      });
      
      // Initialize layout
      slider.style.width = '120px';
    });
  </script>
